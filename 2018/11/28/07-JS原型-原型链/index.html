function Person(){

// Car.prototype.height = 1400;

}

//call在对象中的应用：
//call的根本作用就是改变this指向

//建立构造函数Per
function Per(name,age){ 
    this.name = name;
    this.age = age;
}
//new一个对象per
var per = new Per('deng',18);

//生成一个对象obj
var obj = {
    width : 100,
    height : 200,
};
//调用call()
Per.call(obj,'anna',19);
//它会让Per里面预设的this全变成obj
//最终对象obj拥有两个对象的所有属性
//有点像用Per把对象obj扩展了
// new Per(...)  ==>  Per.call() 就是新建一个per对象
// Per.call(obj, ...);给obj对象添加Per的所有属性
var per2 = new Per('qq',66);
//原来的Per照常使用。


//新的例子：
//一个构造函数Worker
function Worker(){
    this.toWork = function(){
        console.log('working...');
    }
}
//创建对象
var worker = new Worker();
//用Per扩展worker的属性和方法
Per.call(worker,"wang",41);
worker.toWork();
//一个构造函数Bird
function Bird(n){ 
    this.n = n;
    this.tofly = function(){
        console.log(this.n + 'I can fly...');
    }
}
//用Bird再次扩展worker的属性和方法
Bird.call(worker,2);
worker.tofly(2);
worker.name = 'li';
//最后的worker对象除了拥有自己的功能和属性外，
//还拥有Per和Bird的所有功能和属性。
//而原来的Per和Bird构造方法都不变。
var per = new Per('zhang',99);
var bir = new Bird(3);
console.log(per);
console.log(worker);
console.log(bir);
bir.tofly();



//所以最终的继承和扩展写法为：
function Person(name,age,sex){
    this.name = name;
    this.age = age;
    this.sex = sex;
    this.toRun = function(){
        console.log('running...');
    }
}
function Student(name,age,sex,tel,grade){
    // 当new一个对象，时 ===>  var this = {...}
    Person.call(this, name, age, sex);
    //调用Person的call方法，将Person的this指向当前的this，也就实现了Person的功能。
    this.tel = tel;
    this.grade = grade;
}
var stu = new Student('joo',18,'male',13988888000,2017);
stu.toRun();


//标准的分组构造函数
function Component1(arg1,arg2){
    this.arg1 = arg1;
    this.arg2 = arg2;
}
function Component2(arg3,arg4){
    this.arg3 = arg3;
    this.arg4 = arg4;
}
function Component3(arg5,arg6){
    this.arg5 = arg5;
    this.arg6 = arg6;
}
function Complate(arg1,arg2,arg3,arg4,arg5,arg6){
    Component1.call(this,arg1,arg2);
    Component2.call(this,arg3,arg4);
    Component3.call(this,arg5,arg6);
}
var car = new Complate(10,20,30,40,50,60);
console.log(car);
